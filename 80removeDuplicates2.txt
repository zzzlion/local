class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int len = nums.size();
        if(len<=2) return len;
        int k = 2; //[0...k)都是没有重复的
        //如果i扫到的当前元素在k之前已经存在两个（注意，由于A是排好序的，因此只需要判断前两个就行），
        //那么i继续前进。否则将i指向的元素加入k，k与i一起前进。
        for(int i =2;i<len;i++)
        {

           if(nums[i]!=nums[k-2])
            {
                nums[k]=nums[i];
                k++;
            }
                
        }
        return k;
    }
};
